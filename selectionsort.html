Selection Sort

In each loop of selection sort, the minimum element is chosen,
and for loop#1, it is put at the 0th position in the array.
After loop#2, the minimum element is put at the 1st position in the array and so on.
The C++ code for selection sort is 


#include
using namespace std;

void swap(int* int1,int* int2 ){
    int temp=*int1;
    *int1=*int2;
    *int2=temp;
}

void print(int arr[],int n){
    for(int i=0;i<n;i++){
            cout<<arr[i]<<" ";
    }
    cout<<"\n";
}

void selectionsort(int arr[], int size){
    for(int i=0;i<size-1;i++){
        int min_index=i;
        for(int j=i+1; j<size; j++){
            if(arr[j]<arr[min_index]){
                min_index=j;
            }
        }
    swap(&arr[min_index],&arr[i]);
    }
}

int main(){
    int arr[5]={4,5,2,1,7};
    selectionsort(arr,5);
    print(arr,5);
}
