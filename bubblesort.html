Bubble Sort is the simplest sorting algorithm.
It takes an array and compares two elements at a time.
There are two loops running, outer loop and inner loop. At the end of each outer loop, the biggest loop bubbles to the end of the array, hence the name bubble sort.
Here is tht C++ code for bubble sort. 


#include <iostream>
using namespace std;

void swap(int* int1,int* int2 ){
    int temp=*int1;
    *int1=*int2;
    *int2=temp;
}

void bubblesort(int arr[], int len){
int i,j=0;
for(i=0;i<len-1;i++){
    for(j=0;j<len-i-1;j++){
        if(arr[j]>arr[j+1]){
            swap(&arr[j],&arr[j+1]);
        }
    }
}
}

int main()
{
int arr[5]={2,4,3,1};
bubblesort(arr,5);
for(int i=0;i<5;i++){
    cout<<arr[i]<<" ";
}
return 0;
}
