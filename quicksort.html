Quick Sort

Quick sort follows the divide and conquer strategy. We recursively divide the given array in two parts left and right based on a pivot given.
Quicksort makes use of partition function which partitions the given array into two parts.
Firstly there is the pivot element, the element to the left of pivot are smaller than it and elements to the right of pivot are bigger than it.
The C++ code for pivot is given below 


#include<iostream>
using namespace std;

void swap(int* a,int* b){
    int temp=*a;
    *a=*b;
    *b=temp;
}

int partition(int arr[],int start,int end){
    int pivot=arr[end];
    int pindex=start;
    for (int i = start; i<end; i++)
    {
        if(arr[i]<=pivot){
            swap(&arr[i],&arr[pindex]);
            pindex++;
        }

    }
    swap(&arr[pindex],&arr[end]);
    return (pindex);
}

void quicksort(int arr[],int start,int end){

if(start<end){
    int pindex=partition(arr,start,end);
    quicksort(arr,0,pindex-1);
    quicksort(arr,pindex+1,end);
    }
}


int main(){
    int arr[]={5,3,8,1,6,2,7,4};
    quicksort(arr,0,7);
    cout<<"Hello";
    cout<<endl;
    for (int i = 0; i < 8; i++)
    {
        cout<<arr[i]<<" ";
    }
}
