Insertion Sort

In insertion sort, we have a key in array(starting from second element of arrary).
Our target is to move this key to its appropriate position by comparing it if it is smaller than the element previous to it,
or previous to previous to it, we keep on repeating this process it(key) till it becomes bigger than the given element on the array.
Here is the C++ code for Insertion Sort 


#include<iostream>
using namespace std;

void print(int arr[],int n){
    for(int i=0;i<n;i++){
        cout<<arr[i]<<" ";
    }
    cout<<"\n";

}

void insertionsort(int arr[],int n){
int i,j=0;

for(i=1;i<n;i++){
    j=i-1;
    int key=arr[i];
    while(j>=0 && arr[j]>key){
        arr[j+1]=arr[j];
        j--;
    }
    arr[j+1]=key;

}
}

int main(){
int arr[5]={4,5,2,1,7};
insertionsort(arr,5);
print(arr,5);
}
